# Loops

val numbers: List[Int] = List(1, 2, 3, 4)
val a: Int = 0
val b: Int = 5

# For

for (number <- numbers) {
    // do something
}

Nested for loops.

for (i <- List.range(1, 5)) {
    for (j <- List.range(1, 5)) {
        // do something
    }
}

Using for loops and Maps.

val map: Map[String, Int] = Map("Alpha" -> 1, "Beta" -> 2)

for ((k, v) <- map) {
    println(k, v)                       // (Alpha,1) (Beta,2)
}

# While

while (a < b) {
    // do something
}
